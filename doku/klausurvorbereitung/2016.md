```java
public class abc {
    public static int auswerten(Queue upn) {
        Stack stack = new Stack();
        while (!upn.isEmpty()) {
            if (upn.front().equals("+")) {
                int op1 = stack.top;
                stack.pop();
                int op2 = stack.top();
                stack.pop();
                stack.push(op1+op2);
            }
            else if(upn.front().equals("*")) {
                int op1 = stack.top();
                stack.pop();
                int op2 = stack.top();
                stack.pop();
                stack.push(op1*op2);
            }

               stack.push(Integer.parseInt(upn.front()));

        }

    }
}
```
